# 제네릭 인터페이스

## 기본 사용법

```ts
interface KeyPair<T, V> {
  key: T;
  value: V;
}

// 이렇게만하면 에러뜬다 -> KeyPair interface의 매개변수 타입을 지정 해줘야함
let keyPair: KeyPair = {};

// 이런식으로
let keyPair: KeyPair<string, number> = {
  key: "key",
  value: 0,
};

// 그래서 하나의 인터페이스로 다양하게 사용 가능
let KeyPair2: KeyPair<number, string[]>= {
    key:1,
    value:{"잠온다","ㄹㅇㅋㅋ"}
}
```

## 인덱스 시그니처

```ts
// 규칙만 만족하면 어떤 객체든 허용하는
// 아주 유연한 객체 타입을 만드는 문법
interface NumberMap {
  [key: string]: number;
}

// 다 가능~
let numberMap: NumberMap = {
  key: 123,
  key2: -9223,
};

// 제네릭 인터페이스까지 같이 쓰면
interface Map<V> {
  [key: string]: V;
}

let stringMap: Map<string> = {
  key: "value",
};

let booleanMap: Map<boolean> = {
  key: true,
};
```

## 제네릭 타입 별칭

```ts
// 제네릭 인터페이스랑 거의 유사
type Map<V> = {
  [key: string]: V;
};

let stringMap: Map<string> = {
  key: "hello",
};
```

## 활용 예시

- 유저 관리 프로그램
- 유저 구분 : 학생/ 개발자

```ts
// 유저 인터페이스 정의
// 둘은 서로서 유니온 타입
interface Student {
  type: "student";
  school: string;
}

interface Developer {
  type: "developer";
  skill: string;
}

// 공통으로 사용 될 인터페이스
interface User {
  name: string;
  profile: Student | Developer;
}

const developerUser: User = {
  name: "진재환",
  profile: {
    type: "developer",
    skill: "JavaScript",
  },
};

const studentUser: User = {
  name: "최지은",
  profile: {
    type: "student",
    school: "코드스쿼드",
  },
};

function goToSchool(user: User) {
  if (user.profile.type !== "student") {
    console.log("개발자님 잘못오셨습니다");
    return;
  }

  const school = user.profile.school;
  console.log(`${user.name}님 ${school}로 등교완료`);
}

// 이걸 제네릭 인터페이스를 사용하면!
interface User<T> {
  name: string;
  profile: T;
}
interface Student {
  type: "student";
  school: string;
}

interface Developer {
  type: "developer";
  skill: string;
}
const developerUser: User<Developer> = {
  name: "진재환",
  profile: {
    type: "developer",
    skill: "JavaScript",
  },
};

const studentUser: User<Student> = {
  name: "최지은",
  profile: {
    type: "student",
    school: "코드스쿼드",
  },
};
function goToSchool(user: User<Student>) {
  // 제네릭 인터페이스를 통해 타입을 좁혔으니
  // if문 필요없다!
  //   if (user.profile.type !== "student") {
  //     console.log("개발자님 잘못오셨습니다");
  //     return;
  //   }

  const school = user.profile.school;
  console.log(`${user.name}님 ${school}로 등교완료`);
}

// 이제 goToSchool(developer) 을 하면 에러가 뜬다!
function goToWork(user:<Developer>){
```
