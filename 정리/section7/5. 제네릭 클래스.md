# 제네릭 클래스

```ts
class NumberList {
  contructor(private list: number[]) {}

  push(data: number) {
    this.list.push(data);
  }
  pop() {
    this.list.pop();
  }

  print() {
    console.log(this.list);
  }
}

const numberList = new NumberList([1, 2, 3]);
// 아래 함수 모두 정상 작동
numberList.pop();
numberList.push(4);
numberList.print();

// 근데 만약 string List가 필요하다면?
// class 또 만들게??
// 이전에 만들었던 NumberList를 제네릭 클래스로 만들면 됨
class NumberList {
  contructor(private list: number[]) {}

  push(data: number) {
    this.list.push(data);
  }
  pop() {
    this.list.pop();
  }

  print() {
    console.log(this.list);
  }
}
```
